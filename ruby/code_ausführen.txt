Ruby-Console
	irb
		# startet eine Eingabeaufforderung, die alle RUBY-Befehle kennt

Textdatei
	.rb
		# Ruby Code wird in einer Textdatei abgespeichert, die die Endung .rb haben muss, damit diese automatisch
			als ruby erkannt wird (start.rb)

Ausfüren
	ruby start.rb
		# Zum Ausführen wird erst der Interpreter aufgerufen und dahinter die Datei, die ausgeführt werden soll
